### 1.2 内核通信协议

Electron Main Process 与 Kernel Process 之间的 WebSocket 通信协议：

```typescript
// 请求类型
type KernelRequest =
  | { type: 'chat.send'; sessionId: string; agentId: string; message: string }
  | { type: 'session.create'; agentId: string }
  | { type: 'session.list' }
  | { type: 'session.switch'; sessionId: string }
  | { type: 'session.delete'; sessionId: string }
  | { type: 'agent.list' }
  | { type: 'agent.detail'; agentId: string }
  | { type: 'approval.respond'; requestId: string; approved: boolean }
  | { type: 'kernel.shutdown' };

// 响应/事件类型
type KernelEvent =
  | { type: 'delta.text'; sessionId: string; content: string }
  | { type: 'delta.thinking'; sessionId: string; content: string }
  | { type: 'delta.tool_call'; sessionId: string; tool: string; input: unknown }
  | { type: 'tool.started'; sessionId: string; tool: string }
  | { type: 'tool.completed'; sessionId: string; tool: string; output: string }
  | { type: 'approval.request'; sessionId: string; requestId: string; tool: string; input: unknown }
  | { type: 'turn.complete'; sessionId: string; usage: TokenUsage }
  | { type: 'session.created'; sessionId: string }
  | { type: 'session.list'; sessions: SessionInfo[] }
  | { type: 'agent.list'; agents: AgentInfo[] }
  | { type: 'error'; message: string };
```

### 1.3 Agent 配置加密方案

```typescript
// Agent 包格式 (.enc)
interface AgentPackage {
  // AES-256-GCM 加密后的配置
  encrypted: Buffer;
  // 加密 IV
  iv: Buffer;
  // 认证 Tag
  authTag: Buffer;
}

// Agent 明文配置（仅内核运行时解密到内存）
interface AgentConfig {
  id: string;
  version: string;
  identity: {
    name: string;
    nickname: string;
    emoji: string;
    creature: string;
    vibe: string;
    avatar?: string;
  };
  soul: string;           // SOUL.md 内容（系统提示词核心）
  agents: string;         // AGENTS.md 内容（工作手册）
  tools: string;          // TOOLS.md 内容（工具配置）
  user?: string;          // USER.md 内容（用户画像）
  heartbeat?: string;     // HEARTBEAT.md 内容
  toolWhitelist?: string[];   // 工具白名单
  toolBlacklist?: string[];   // 工具黑名单
  maxTurns: number;       // 最大推理轮次
  model?: string;         // 模型覆盖
  temperature?: number;   // 温度覆盖
}

// Agent 广场展示信息（不加密，供 UI 使用）
interface AgentManifestEntry {
  id: string;
  name: string;
  nickname: string;
  emoji: string;
  creature: string;
  vibe: string;
  description: string;
  tags: string[];
  scenarios: string[];
  version: string;
}

```

AGENTS.md                   →    config.agents
TOOLS.md                    →    config.tools
USER.md                     →    config.user (可选)
HEARTBEAT.md                →    config.heartbeat (可选)
docs/workflows/*.md         →    合并到 config.soul 或 config.agents

### 6.1 ReAct 循环引擎（核心）

```typescript
// kernel/src/engine/react-loop.ts
async function* runReActLoop(
  provider: AIProvider,
  messages: Message[],
  tools: ToolDefinition[],
  agentConfig: AgentConfig,
  maxTurns: number = agentConfig.maxTurns ?? 64
): AsyncGenerator<KernelEvent> {
  for (let turn = 0; turn < maxTurns; turn++) {
    // 1. 调用 LLM（流式）
    yield { type: 'turn.started', turn };

    const stream = provider.streamMessage({
      messages,
      tools,
      system: buildSystemPrompt(agentConfig),
    });

    let assistantContent = '';
    const toolCalls: ToolCall[] = [];

    for await (const event of stream) {
      if (event.type === 'text_delta') {
        assistantContent += event.text;
        yield { type: 'delta.text', content: event.text };
      }
      if (event.type === 'tool_call_delta') {
        // 收集工具调用
      }
    }

    messages.push({ role: 'assistant', content: assistantContent, toolCalls });

    // 2. 如果没有工具调用，循环结束
    if (toolCalls.length === 0) {
      yield { type: 'turn.complete', usage: stream.usage };
      return;
    }

    // 3. 执行工具调用
    for (const call of toolCalls) {
      yield { type: 'tool.started', tool: call.name };

      // 权限检查
      const approved = await checkPermission(call, agentConfig);
      if (!approved) {
        yield { type: 'approval.request', tool: call.name, input: call.input };
        // 等待用户审批...
      }

      const result = await executeTool(call);
      yield { type: 'tool.completed', tool: call.name, output: result };
      messages.push({ role: 'tool', toolCallId: call.id, content: result });
    }
  }
}
```

### 6.2 内核进程生命周期管理

```typescript
// electron/extensions/kernel/kernel-launcher.ts
class KernelLauncher {
  private process: ChildProcess | null = null;
  private wsClient: WebSocket | null = null;
  private port: number = 0;

  async start(): Promise<void> {
    // 1. 找到内核二进制
    const kernelBin = this.resolveKernelBinary();

    // 2. 启动进程
    this.process = spawn(kernelBin, ['--port', '0'], {
      stdio: ['pipe', 'pipe', 'pipe'],
    });

    // 3. 读取动态端口
    this.port = await this.readPortFromStdout(this.process);

    // 4. 建立 WS 连接
    this.wsClient = new WebSocket(`ws://127.0.0.1:${this.port}`);

    // 5. 开始心跳监控
    this.startHeartbeat();
  }

  async stop(): Promise<void> {
    this.wsClient?.send(JSON.stringify({ type: 'kernel.shutdown' }));
    // 等待优雅关闭，超时后 kill
  }

  private startHeartbeat(): void {
    setInterval(() => {
      if (!this.wsClient || this.wsClient.readyState !== WebSocket.OPEN) {
        this.restart();
      }
    }, 5000);
  }
}
```

### 6.3 Agent 配置加密

```typescript
// kernel/src/agent/agent-loader.ts
import { createDecipheriv, createCipheriv, randomBytes } from 'crypto';

const ALGORITHM = 'aes-256-gcm';

function encryptAgentConfig(config: AgentConfig, key: Buffer): AgentPackage {
  const iv = randomBytes(16);
  const cipher = createCipheriv(ALGORITHM, key, iv);

  const plaintext = JSON.stringify(config);
  const encrypted = Buffer.concat([
    cipher.update(plaintext, 'utf8'),
    cipher.final(),
  ]);

  return {
    encrypted,
    iv,
    authTag: cipher.getAuthTag(),
  };
}

function decryptAgentConfig(pkg: AgentPackage, key: Buffer): AgentConfig {
  const decipher = createDecipheriv(ALGORITHM, key, pkg.iv);
  decipher.setAuthTag(pkg.authTag);

  const decrypted = Buffer.concat([
    decipher.update(pkg.encrypted),
    decipher.final(),
  ]);

  return JSON.parse(decrypted.toString('utf8'));
}
```
